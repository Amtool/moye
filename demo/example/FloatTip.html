<button id="floattip-dispose">销毁</button>

require(['FloatTip'], function (FloatTip) {
  var floattip = new FloatTip({
      content: '弹出提示层',
      left: '50%',
      top: '250px',
      fixed: 0
  });

  floattip.render();

  $('#floattip-show').on('click', function() {
      floattip.show();
  });

  $('#floattip-hide').on('click', function() {
      floattip.hide();
  });

  $('#floattip-dispose').on('click', function() {
      floattip.dispose();
  });

});
